# NAME

crmd - File/Directory Handler about Copy, Rename, Move, Delete.

# SYNOPSIS

    # rcopy
    $ perl copy-rename.pl rcopy

    # rname
    $ perl copy-rename.pl rname

    # rmove
    $ perl move.pl

    # delete
    $ perl delete.pl
